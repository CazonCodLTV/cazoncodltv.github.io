from pynput import keyboard as kb
import random
import pyperclip
import time

texts = ["text1", "text2", "text3", "text4"]

print("Running... Press F4 to stop.")

stop_flag = False

def on_press(key):
    global stop_flag
    if key == kb.Key.f4:
        stop_flag = True
        return False

listener = kb.Listener(on_press=on_press)
listener.start()

try:
    while True:
        if stop_flag:
            print("Stopped.")
            break
        chosen = random.choice(texts)
        pyperclip.copy(chosen)
        time.sleep(0.1)
        # simulate typing
        for ch in chosen:
            kb.Controller().type(ch)
        time.sleep(0.5)
except KeyboardInterrupt:
    print("Stopped manually.")")
