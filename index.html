user32 = ctypes.WinDLL('user32', use_last_error=True)

print("Running... Press F4 to stop.")

STOP_KEY = 0x73  # F4 virtual-key code

def is_key_pressed(vk):
    return user32.GetAsyncKeyState(vk) & 0x8000
