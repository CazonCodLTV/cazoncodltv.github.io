import ctypes
import time
import random

# Random strings to paste
texts = ["text1", "text2", "text3", "text4"]

# Windows key constants
VK_CONTROL = 0x11
VK_V = 0x56
VK_F4 = 0x73

# GetAsyncKeyState reads if a key is pressed
GetKeyState = ctypes.windll.user32.GetAsyncKeyState

# Function to simulate Ctrl+V
def ctrl_v():
    ctypes.windll.user32.keybd_event(VK_CONTROL, 0, 0, 0)
    ctypes.windll.user32.keybd_event(VK_V, 0, 0, 0)
    ctypes.windll.user32.keybd_event(VK_V, 0, 2, 0)
    ctypes.windll.user32.keybd_event(VK_CONTROL, 0, 2, 0)

# Function to copy a chosen text to clipboard
def set_clipboard(text):
    ctypes.windll.user32.OpenClipboard(0)
    ctypes.windll.user32.EmptyClipboard()
    ctypes.windll.user32.SetClipboardData(1, ctypes.c_wchar_p(text))
    ctypes.windll.user32.CloseClipboard()

print("Running... Press F4 to stop.")

while True:
    # Stop when F4 is pressed
    if GetKeyState(VK_F4):
        print("Stopped.")
        break

    # Pick a random text
    choice = random.choice(texts)

    # Put it into clipboard
    set_clipboard(choice)

    # Paste it
    ctrl_v()

    time.sleep(0.05)   # small delay (optional)
